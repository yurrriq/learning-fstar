Type Synonyms¶

Define a type synonym to model filenames as strings.

(** A filename is a string. *)
type filename = string

Access Control Predicates¶

For this exercise, the file demo/tempfile is writeable and all others are not.

(** Determine whether a file can be written. *)
val canWrite : filename -> Tot bool
let canWrite = function
  | "demo/tempfile" -> true
  | _               -> false

A file is considered readable if it either satisfies canWrite or is demo/README.

(** Determine if a file can be read. *)
val canRead : filename -> Tot bool
let canRead (f:filename) =
  canWrite f ||
  f = "demo/README"
